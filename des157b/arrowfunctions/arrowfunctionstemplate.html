Named Function

The function below prompts the user to choose a color from the options provided, which are actually variables in an array. When the user chooses a color the loop in the code will change the background color of the page.

            function question(){
                var whatColor = prompt("Enter your favorite color from the list: red, yellow, blue, or green", "");
                var color = ["red","blue","yellow","green"];
                
                if (whatColor == "red"){
                   document.body.style.backgroundColor = color[0];
                }
                else if(whatColor == "blue"){
                   document.body.style.backgroundColor = color[1];
               }
               else if(whatColor == "yellow"){
                   document.body.style.backgroundColor = color[2];
               }
               else if(whatColor == "green"){
                   document.body.style.backgroundColor = color[3];
               }
               }
               question();
